import 'package:flutter/material.dart';
import 'package:flutter_flashcards/components/app/custom_appbar.dart';
import 'package:flutter_flashcards/components/review_page/header_button.dart';
import 'package:flutter_flashcards/configs/constants.dart';

import '../components/review_page/word_tile.dart';
import '../models/word.dart';

class ReviewPage extends StatefulWidget {
  const ReviewPage({Key? key}) : super(key: key);
  @override
  State<ReviewPage> createState() => _ReviewPageState();
}

class _ReviewPageState extends State<ReviewPage> {

  final _listKey = GlobalKey<AnimatedListState>();

  List<Word> _reviewWords = [

  ];

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: PreferredSize(
          preferredSize: Size.fromHeight(kAppBarHeight),
          child: CustomAppBar()),
      body: Column(
        children: [
          Expanded(
            child: Row(
              children: [
                HeaderButton(title: 'Insert', onTap: (){
                  _insertWord(word:
                  Word(topic: "City", english: "Hospital", pinyin: "Yī Yuàn", character: "医院"),);
                },),
                HeaderButton(title: 'Insert List', onTap: (){
                  final _wordsToAdd = [
                    Word(topic: "Drinks", english: "Juice", pinyin: "Guǒ Zhī", character: "果汁"),
                    Word(topic: "City", english: "Hospital", pinyin: "Yī Yuàn", character: "医院"),
                    Word(topic: "School", english: "School Bag", pinyin: "Shū Bāo", character: "书包"),
                    Word(topic: "Shapes", english: "Triangle", pinyin: "Sān Jiǎo Xíng", character: "三角形"),
                    Word(topic: "Insects", english: "Grasshopper", pinyin: "Zhà Měng", character: "蚱蜢"),
                    Word(topic: "Weather", english: "Rain", pinyin: "Xià Yǔ", character: "下雨"),
                  ];
                  _insertWords(words: _wordsToAdd);
                },),
                HeaderButton(title: 'Clear', onTap: (){
                  _clearAllWords();
                },),
              ],
            ),
          ),
          Expanded(
              flex: 10,
              child: AnimatedList(
                key: _listKey,
                initialItemCount: _reviewWords.length,
                itemBuilder: (context, index, animation) =>
                    WordTile(word: _reviewWords[index], animation: animation, onTapped: (){
                      _removeWord(word: _reviewWords[index]);
                    },),
              ))
        ],
      ),
    );
  }

  _insertWord({required Word word}){
    _listKey.currentState?.insertItem(_reviewWords.length);
    _reviewWords.insert(_reviewWords.length, word);
  }

  _insertWords({required List<Word> words}){
    for(int i = 0; i < words.length; i++){
      _listKey.currentState?.insertItem(i);
      _reviewWords.add(words[i]);
    }

  }
  
  _removeWord({required Word word}){
    _listKey.currentState?.removeItem(_reviewWords.indexOf(word),
            duration: Duration(milliseconds: 5000),
            (context, animation) => WordTile(
        word: word, animation: animation, onTapped: (){},));
    _reviewWords.remove(word);
  }

  _clearAllWords(){
    for(int i = 0; i < _reviewWords.length; i++){
      _listKey.currentState?.removeItem(0, (context, animation) => WordTile(
          word: _reviewWords[i], animation: animation, onTapped: (){}));
    }
    WidgetsBinding.instance.addPostFrameCallback((timeStamp) {
      _reviewWords.clear();
    });

  }
}
